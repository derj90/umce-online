2 — Arte generativo p5.js por tipo

Cada tipo tiene un patrón generativo para fondos de portada y pantallas de completado. Se instancia con createCourseCanvas(type, containerId). Paleta como config, forma parametrizable. IMPORTANTE: Solo usar ellipse(), line(), rect(), vertex() — NO curveVertex() ni bezierVertex() (crashean sin error visible en producción).

/* ================================================================ GENERATIVE ART ENGINE — curso-canvas.js Instrucciones de seguridad p5.js (ver feedback_p5_canvas_react.md): - NUNCA curveVertex / bezierVertex - Solo ellipse / line / vertex / rect simples - Sin frameRate() > 30 en móvil ================================================================ */ var COURSE_CANVAS_CONFIG = { autoformacion: { palette: ['#6366f1','#a5b4fc','#14b8a6','#5eead4','#e0e7ff'], bg: '#f5f6ff', fn: 'floatingOrbs', // orbs suaves, movimiento lento, mouse repulsion speed: 0.4, count: 18, opacity: 0.12 }, tutoreado: { palette: ['#4338ca','#7c3aed','#c4b5fd','#312e81'], bg: '#f3f2ff', fn: 'geometricLattice', // red de líneas + nodos, respiración lenta speed: 0.2, count: 12, opacity: 0.10 }, diplomado: { palette: ['#0f766e','#14b8a6','#d97706','#fbbf24','#99f6e4'], bg: '#f0fdf9', fn: 'waveLines', // ondas horizontales, movimiento continuo speed: 0.5, count: 8, opacity: 0.14 }, taller: { palette: ['#f97316','#dc2626','#fed7aa','#fca5a5'], bg: '#fff8f3', fn: 'energyParticles', // partículas rápidas que se dispersan speed: 1.2, count: 30, opacity: 0.09 }, induccion: { palette: ['#4f9647','#78b367','#ca8a04','#fef08a','#cfe8c4'], bg: '#f7faf5', fn: 'floatingLeaves', // hojas simples (ellipses rotadas), caída lenta speed: 0.3, count: 20, opacity: 0.15 } }; function createCourseCanvas(type, containerId) { var cfg = COURSE_CANVAS_CONFIG[type] || COURSE_CANVAS_CONFIG.autoformacion; var sketch = function(p) { var particles = []; var W, H; p.setup = function() { var el = document.getElementById(containerId); W = el.offsetWidth; H = el.offsetHeight; var cnv = p.createCanvas(W, H); cnv.parent(containerId); p.frameRate(24); // máx 24fps para rendimiento for (var i = 0; i < cfg.count; i++) particles.push(newParticle(p, cfg, W, H)); }; p.draw = function() { p.clear(); p.background(cfg.bg); particles.forEach(function(pt) { updateParticle(p, pt, cfg, W, H); drawParticle(p, pt, cfg); }); }; // Repulsión de mouse (igual a sustentabilidad2026) p.mouseMoved = function() { particles.forEach(function(pt) { var dx = p.mouseX - pt.x, dy = p.mouseY - pt.y; var d = Math.sqrt(dx*dx + dy*dy); if (d < 80) { pt.vx -= (dx / d) * 0.8; pt.vy -= (dy / d) * 0.8; } }); }; }; return new p5(sketch); } function newParticle(p, cfg, W, H) { return { x: p.random(W), y: p.random(H), vx: p.random(-cfg.speed, cfg.speed), vy: p.random(-cfg.speed * 0.5, cfg.speed), r: p.random(8, 40), color: cfg.palette[Math.floor(p.random(cfg.palette.length))], rot: p.random(p.TWO_PI) }; } function updateParticle(p, pt, cfg, W, H) { pt.x += pt.vx; pt.y += pt.vy; pt.vx *= 0.99; pt.vy *= 0.99; pt.rot += 0.005; if (pt.x < -50) pt.x = W + 50; if (pt.x > W + 50) pt.x = -50; if (pt.y < -50) pt.y = H + 50; if (pt.y > H + 50) pt.y = -50; } function drawParticle(p, pt, cfg) { p.push(); p.translate(pt.x, pt.y); p.rotate(pt.rot); p.noStroke(); var c = p.color(pt.color); c.setAlpha(cfg.opacity * 255); p.fill(c); // Solo ellipse — no bezier/curve p.ellipse(0, 0, pt.r * 2, pt.r * 1.4); p.pop(); }

3c — Contenido del slide (prose-edu)

Biodiversidad y ecosistemas

Cuando se estudia el funcionamiento de la biodiversidad en un ecosistema, se analizan tres atributos fundamentales: composición, estructura y función. La pérdida de cualquiera compromete la capacidad del sistema para sostener la vida.

La diversidad biológica es la base sobre la cual se construyen todos los servicios ecosistémicos que la humanidad necesita para sobrevivir.

Los tres atributos

  • Composición: qué especies están presentes
  • Estructura: cómo se organizan espacialmente
  • Función: qué procesos ecológicos realizan
